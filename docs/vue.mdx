### Vue Component with Composition API

```.js
<template>
  <button @click="onClick">
    Count is: {{ count }}, double is: {{ double }}, click to increase it.
  </button>
</template>

<script>
  import { ref, computed, onMounted } from 'vue'

  export default {
    setup() {
      onMounted(() => console.log('component mounted!'))

      const count = ref(0)
      const onClick = () => {
        count.value++
      }
      const double = computed(() => count.value * 2)

      return {
        count,
        double,
        onClick
      }
    }
  }
</script>

<style scoped>button {}</style>
```

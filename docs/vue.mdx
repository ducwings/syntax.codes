## Vue 3

### Functional Component

```.js
<template>
  <button @click="increment">
    Count is: {{ count }}, double is: {{ double }}, click to increase it.
  </button>
</template>

<script>
import { ref, computed, onMounted } from 'vue'

export default {
  setup() {
    onMounted(() => console.log('component mounted!'))

    const count = ref(0)
    const increment = () => {
      count.value++
    }
    const double = computed(() => count.value * 2)

    return {
      count,
      double,
      increment
    }
  }
}
</script>

<style scoped></style>
```

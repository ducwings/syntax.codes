## Angular 9.1

### Angular Component with Typescript

```.js
import { Component, OnInit } from '@angular/core';
interface Item {
  id: number;
}

@Component({
  selector: 'app-root',
  template: `
    <ul>
      <li *ngFor="let item of collection; trackBy: trackByFn">Item {{ item.id }}</li>
    </ul>
    <button (click)="fetchItems()">Fetch items</button>
  `,
  // templateUrl: './MyComponent.html', // or use template file
  styleUrls: ['./app.component.css']
})
export class MyComponent implements OnInit, OnDestroy {
  collection: Item[] = [{ id: 1 }, { id: 2 }];

  ngOnInit(): void {}
  ngOnDestroy(): void {}

  fetchItems(): void {
    this.collection = [{ id: 1 }, { id: 2 }, { id: 3 }, { id: 4 }];
  }

  trackByFn(index: number, item: Item) {
    return item.id;
  }
}
```

## Stencil 1.14

### Stencil Component

```.js
import { Component, State, Listen, h } from '@stencil/core';
  
@Component({
  tag: 'my-toggle-button'
})
export class MyToggleButton {
  @State() open: boolean;
  
  @Listen('click', { capture: true })
  handleClick() {
    this.open = !this.open;
  }
  
  render() {
    return <button>
      {this.open ? "On" : "Off"}
    </button>;
  }
}
```
